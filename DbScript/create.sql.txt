DROP TABLE IF EXISTS phone;
DROP TABLE IF EXISTS appuser;
DROP TABLE IF EXISTS app_user;
 
CREATE TABLE appuser (
  appuserid varchar(37),
  firstname VARCHAR(250) NOT NULL,
  lastname VARCHAR(250) NOT NULL,
  mail VARCHAR(250) NOT NULL UNIQUE,
  password VARCHAR(250) NOT NULL,
  created timestamp,
  modified timestamp,
  lastlogin timestamp,
  isactive boolean,
  token varchar(250),
  primary key(appuserid)
);

CREATE TABLE phone (
  phoneid BIGINT GENERATED BY DEFAULT AS IDENTITY,
  number VARCHAR(250) NOT NULL,
  citycode VARCHAR(20) NOT NULL,
  countrycode VARCHAR(250) NOT NULL,
  appuserid VARCHAR(37),
  primary key(phoneid),
  foreign key(appuserid) references appuser(appuserid)
);
